const tab = link.getAttribute('data-tab'); switchTab(tab); }); }); } function switchTab(tab) { if (isProcessing) return; // Don't allow tab switching during processing activeTab = tab; // Update nav links document.querySelectorAll('.nav-link').forEach(link => { link.classList.remove('active'); }); document.querySelector(`[data-tab="${tab}"]`).classList.add('active'); // Update tab content document.querySelectorAll('.tab-content').forEach(content => { content.classList.remove('active'); }); document.getElementById(`${tab}Tab`).classList.add('active'); // Update metrics dashboard if (tab === 'compress') { document.getElementById('compressMetrics').style.display = 'grid'; document.getElementById('decompressMetrics').style.display = 'none'; } else { document.getElementById('compressMetrics').style.display = 'none'; document.getElementById('decompressMetrics').style.display = 'grid'; } } // Decompression functionality function setupDecompression() { const decompressZone = document.getElementById('decompressZone'); const decompressInput = document.getElementById('decompressInput'); decompressZone.addEventListener('click', () => decompressInput.click()); decompressZone.addEventListener('dragover', handleDecompressDragOver); decompressZone.addEventListener('dragleave', handleDecompressDragLeave); decompressZone.addEventListener('drop', handleDecompressDrop); decompressInput.addEventListener('change', handleDecompressFileSelect); } function handleDecompressDragOver(e) { e.preventDefault(); e.stopPropagation(); document.getElementById('decompressZone').classList.add('dragover'); } function handleDecompressDragLeave(e) { e.preventDefault(); e.stopPropagation(); document.getElementById('decompressZone').classList.remove('dragover'); } function handleDecompressDrop(e) { e.preventDefault(); e.stopPropagation(); document.getElementById('decompressZone').classList.remove('dragover'); const files = Array.from(e.dataTransfer.files); processDecompressFiles(files); } function handleDecompressFileSelect(e) { const files = Array.from(e.target.files); processDecompressFiles(files); } async function processDecompressFiles(files) { if (isProcessing || files.length === 0) return; // Validate file types const invalidFiles = files.filter(file => !file.name.endsWith('.qf')); if (invalidFiles.length > 0) { showErrorNotification(`Invalid file format: ${invalidFiles.map(f => f.name).join(', ')}. Only .qf files are supported for decompression.`); return; } isProcessing = true; for (const file of files) { await decompressFile(file); } isProcessing = false; } async function decompressFile(file) { const progressContainer = document.getElementById('decompressProgressContainer'); const progressFill = document.getElementById('decompressProgressFill'); const progressPercentage = document.getElementById('decompressProgressPercentage'); const progressTitle = document.getElementById('decompressProgressTitle'); const processingOverlay = document.getElementById('decompressProcessingOverlay'); const processingText = document.getElementById('decompressProcessingText'); const processingSubtext = document.getElementById('decompressProcessingSubtext'); progressContainer.classList.add('visible'); processingOverlay.classList.add('visible'); let currentPhase = 0; let overallProgress = 0; // Process through decompression phases for (let i = 0; i < decompressPhases.length; i++) { const phase = decompressPhases[i]; currentPhase = i; // Update phase indicators updateDecompressPhaseIndicators(i); // Update processing text processingText.textContent = phase.name; processingSubtext.textContent = getDecompressPhaseDescription(phase.id); // Simulate phase processing const phaseStart = Date.now(); while (Date.now() - phaseStart < phase.duration) { const phaseProgress = (Date.now() - phaseStart) / phase.duration; overallProgress = (i + phaseProgress) / decompressPhases.length * 100; progressFill.style.width = overallProgress + '%'; progressPercentage.textContent = Math.round(overallProgress) + '%'; await new Promise(resolve => setTimeout(resolve, 50)); } // Mark phase as completed document.getElementById(`decompress-phase-${phase.id}`).classList.add('completed'); } // Complete decompression progressFill.style.width = '100%'; progressPercentage.textContent = '100%'; // Simulate decompression result const originalSize = file.size * (2 + Math.random() * 3); // Simulate expansion const decompressionTime = 800 + Math.random() * 2000; const quantumIntegrity = 0.95 + Math.random() * 0.05; const decompressionResult = { originalFileName: file.name.replace('.qf', ''), compressedSize: file.size, decompressedSize: originalSize, decompressionTime, quantumIntegrity, entanglementPairsRestored: Math.floor(Math.random() * 100), superpositionStatesCollapsed: Math.floor(Math.random() * 500), interferencePatternsMapped: Math.floor(Math.random() * 200), timestamp: Date.now() }; decompressionHistory.push(decompressionResult); filesRestored++; // Update metrics updateDecompressionMetrics(decompressionResult); // Simulate download setTimeout(() => { const link = document.createElement('a'); link.href = '#'; link.download = decompressionResult.originalFileName; document.body.appendChild(link); link.click(); document.body.removeChild(link); showSuccessNotification(`${file.name} decompressed successfully!`); }, 1000); // Hide overlays after delay setTimeout(() => { progressContainer.classList.remove('visible'); processingOverlay.classList.remove('visible'); resetDecompressPhaseIndicators(); }, 3000); } function updateDecompressPhaseIndicators(currentPhase) { decompressPhases.forEach((phase, index) => { const dot = document.getElementById(`decompress-phase-${phase.id}`); if (index < currentPhase) { dot.classList.add('completed'); dot.classList.remove('active'); } else if (index === currentPhase) { dot.classList.add('active'); dot.classList.remove('completed'); } else { dot.classList.remove('active', 'completed'); } }); } function resetDecompressPhaseIndicators() { decompressPhases.forEach(phase => { const dot = document.getElementById(`decompress-phase-${phase.id}`); dot.classList.remove('active', 'completed'); }); } function getDecompressPhaseDescription(phaseId) { const descriptions = { 'read': 'Extracting quantum metadata...', 'reconstruct': 'Restoring entanglement pairs...', 'analyze': 'Mapping interference patterns...', 'collapse': 'Collapsing superposition states...', 'restore': 'Reconstructing original data...' }; return descriptions[phaseId] || 'Processing...'; } function updateDecompressionMetrics(result) { document.getElementById('quantumIntegrity').textContent = (result.quantumIntegrity * 100).toFixed(1) + '%'; document.getElementById('expansionRatio').textContent = (result.decompressedSize / result.compressedSize).toFixed(1) + 'x'; document.getElementById('decompressionTime').textContent = formatTime(result.decompressionTime); document.getElementById('filesRestored').textContent = filesRestored; // Update change indicators document.getElementById('integrityChange').textContent = (result.quantumIntegrity * 100).toFixed(1) + '%'; document.getElementById('expansionChange').textContent = (result.decompressedSize / result.compressedSize).toFixed(1) + 'x'; document.getElementById('decompressTimeChange').textContent = formatTime(result.decompressionTime); document.getElementById('restoredChange').textContent = '+' + filesRestored; } function formatTime(ms) { if (ms < 1000) return Math.round(ms) + 'ms'; return (ms / 1000).toFixed(1) + 's'; } function showSuccessNotification(message) { const notification = document.createElement('div'); notification.style.cssText = ` position: fixed; top: 2rem; right: 2rem; background: linear-gradient(135deg, #10b981, #059669); color: white; padding: 1rem 1.5rem; border-radius: 12px; box-shadow: 0 10px 25px rgba(16, 185, 129, 0.3); z-index: 10000; font-weight: 600; max-width: 400px; animation: slideInRight 0.3s ease; `; notification.textContent = message; document.body.appendChild(notification); setTimeout(() => { notification.style.animation = 'slideOutRight 0.3s ease'; setTimeout(() => { if (notification.parentNode) { notification.parentNode.removeChild(notification); } }, 300); }, 3000); } function showErrorNotification(message) { const notification = document.createElement('div'); notification.style.cssText = ` position: fixed; top: 2rem; right: 2rem; background: linear-gradient(135deg, #ef4444, #dc2626); color: white; padding: 1rem 1.5rem; border-radius: 12px; box-shadow: 0 10px 25px rgba(239, 68, 68, 0.3); z-index: 10000; font-weight: 600; max-width: 400px; animation: slideInRight 0.3s ease; `; notification.textContent = message; document.body.appendChild(notification); setTimeout(() => { notification.style.animation = 'slideOutRight 0.3s ease'; setTimeout(() => { if (notification.parentNode) { notification.parentNode.removeChild(notification); } }, 300); }, 5000); }
